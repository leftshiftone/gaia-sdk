mutation {
    @IncomingType
    impulse {
       impulsePayload:byte[]
       impulseHeader {
          identityId:string
          clientId:string
          userId:string
       }
    }
    @IncomingType
    impulseContext {
       impulseContextPayload:byte[]
       impulseContextHeader {
          identityId:string
          clientId:string
          userId:string
       }
    }
    @IncomingType
    impulseNotification {
       impulseNotificationPayload:byte[]
       impulseNotificationHeader {
          identityId:string
          clientId:string
          userId:string
       }
    }
    @IncomingType
    impulseLog {
       impulseLogPayload:byte[]
       impulseLogHeader {
          identityId:string
          clientId:string
          userId:string
       }
    }

    dispatchImpulse(impulse:impulse):boolean
    dispatchImpulseContext(impulse:impulseContext):boolean
    dispatchImpulseNotification(impulse:impulseNotification):boolean
    dispatchImpulseLog(impulse:impulseLog):boolean
}
